  def self.getresults( sp, echo = true )
    results = ""
    while (c = sp.getc) do
      if ( !echo.nil? && echo )
        putc c
      end
      results += "#{c.chr}"
    end

    # deal with multiple lines
    results.gsub!( "\r", "\n")
  end

  def initialize( xbee_usbdev_str, baud, data_bits, stop_bits, parity )
    # open serial port device to XBee
    @xbee = SerialPort.new( xbee_usbdev_str, baud.to_i, data_bits.to_i, stop_bits.to_i, parity )
    @xbee.read_timeout = TYPICAL_READ_TIMEOUT
    @baudcodes = { 1200 => 0, 2400 => 1, 4800 => 2, 9600 => 3, 19200 => 4, 38400 => 5, 57600 => 6, 115200 => 7 }
    @paritycodes = { :None => 0, :Even => 1, :Odd => 2, :Mark => 3, :Space => 4 }
    @iotypes = { :Disabled => 0, :ADC => 2, :DI => 3, :DO_Low => 4, :DO_High => 5, 
                 :Associated_Indicator => 1, :RTS => 1, :CTS => 1, :RS485_Low => 6, :RS485_High => 7 }

dio ( port )

reads an i/o port configuration on the XBee for analog to digital or digital input or output (GPIO)

this method returns an I/O type symbol of: 

:Disabled
:ADC
:DI
:DO_Low
:DO_High
:Associated_Indicator
:RTS
:CTS
:RS485_Low
:RS485_High

Not all DIO ports are capable of every configuration listed above. This method will properly translate the XBee’s response value to the symbol above when the same value has different meanings from port to port.

The port parameter may be any symbol :D0 through :D8 representing the 8 I/O ports on an XBee

[show source] 
# File xbee.rb, line 409
  def dio( port )
    at = "AT#{port.to_s}\r"
    @xbee.write( at )
    response = getresponse.to_i

    if response == 1  # the value of 1 is overloaded based on port number
      case port 
      when :D5 
        return :Associated_Indicator
      when :D6
        return :RTS
      when :D7
        return :CTS
      end
    else
      @iotypes.index(response)
    end

  end

io_input ()

Forces a sampling of all DIO pins configured for input via dio! Returns a hash with the following key/value pairs: :NUM => number of samples :CM => channel mask :DIO => dio data if DIO lines are enabled :ADCn => adc sample data (one for each ADC channel enabled)

[show source] 
# File xbee.rb, line 496
  def io_input

    tmp = @xbee.read_timeout 
    @xbee.read_timeout = LONG_READ_TIMEOUT

    @xbee.write("ATIS\r")
    response = getresponse
    linenum = 1
    adc_sample = 1
    samples = Hash.new

    if response.match("ERROR") 
      samples[:ERROR] = "ERROR"
      return samples  
    end

    # otherwise parse input data
    response.each_line do | line |
      case linenum 
      when 1
        samples[:NUM] = line.to_i 
      when 2
        samples[:CM] = line.strip.chomp
      when 3
        samples[:DIO] = line.strip.chomp
      else
        sample = line.strip.chomp
        if ( !sample.nil? && sample.size > 0 )
          samples["ADC#{adc_sample}".to_sym] = line.strip.chomp
          adc_sample += 1
        end
      end

      linenum += 1
    end

    @xbee.read_timeout = tmp
    samples
  end

neighbors ()

Neighbor node discovery. Returns an array of hashes each element of the array contains a hash each hash contains keys: :MY, :SH, :SL, :DB, :NI representing addresses source address, Serial High, Serial Low, Received signal strength, node identifier respectively. Aan example of the results returned (hash as seen by pp): 

[{:NI=>" ", :MY=>"0", :SH=>"13A200", :SL=>"4008A642", :DB=>-24},
 {:NI=>" ", :MY=>"0", :SH=>"13A200", :SL=>"4008A697", :DB=>-33},
 {:NI=>" ", :MY=>"0", :SH=>"13A200", :SL=>"40085AD5", :DB=>-52}]

Signal strength (:DB) is reported in units of -dBM.

[show source] 
# File xbee.rb, line 112
  def neighbors

    # neighbors often takes more than 1000ms to return data
    tmp = @xbee.read_timeout 
    @xbee.read_timeout = LONG_READ_TIMEOUT
    @xbee.write("ATND\r")
    response = getresponse  

    # parse nodes and stuff an array of hashes
    @neighbors = Array.new
    linetype = 0
    neighbor = 0

    if response.nil?
      return @neighbors   # return an empty array
    end

    response.each_line do | line |
     
      line.chomp!

      if line.size > 0  
        case linetype
        when 0    # MY
            @neighbors[ neighbor ] = Hash.new
            @neighbors[ neighbor ].store( :MY, line )

        when 1    # SH
            @neighbors[ neighbor ].store( :SH, line )

        when 2    # SL
            @neighbors[ neighbor ].store( :SL, line )

        when 3    # DB
            @neighbors[ neighbor ].store( :DB, -(line.hex) )

        when 4    # NI
            @neighbors[ neighbor ].store( :NI, line )

            neighbor += 1
        end

        if linetype < 4
          linetype += 1
        else
          linetype = 0
        end
      end
    end

    @xbee.read_timeout = tmp
    @neighbors
  end
