#!/bin/bash

RELEASE=ruby-xbee-1.0

pushd . mkdir -p releases/$RELEASE rm -rf releases/tmp mkdir -p releases/tmp/$RELEASE cp -vr { README RELEASE conf doc license ruby-serialport-0.6-linux-i386 ruby-serialport-0.6-osx-intel ruby-serialport-0.6-ppc xbee.rb xbeeconfigure.rb xbeedio.rb xbeeinfo.rb xbeelisten.rb xbeesend.rb } releases/tmp/$RELEASE echo DONE WITH COPY cd releases/tmp/$RELEASE

echo REMOVING .svn artifacts from pwd find . -name “.svn” -exec rm -rf {} ; cd .. pwd echo creating TARBALL tar cvfz ../../releases/$RELEASE/$RELEASE.tar.gz $RELEASE md5 ../../releases/$RELEASE/$RELEASE.tar.gz popd

# clean up tmp directory rm -rf releases/tmp
